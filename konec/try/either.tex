\def\Alter#1#2{{%
	\setbox\@tempboxa\hbox{\footnotesize #1}%
	\@tempdima\wd\@tempboxa%
	\setbox\@tempboxa\hbox{\footnotesize #2}%
	\@tempdimb\wd\@tempboxa%
	\ifnum\@tempdima<\@tempdimb
		\hbox to \@tempdimb{%
			\lower.8ex\hbox to 0pt{\footnotesize #2\hss}%
			\hfil\raise.8ex\hbox{\footnotesize #1}\hfil}%
	\else
		\hbox to \@tempdima{%
			\raise.8ex\hbox to 0pt{\footnotesize #1\hss}%
			\hfil\lower.8ex\hbox{\footnotesize #2}\hfil}%
	\fi}}

\def\spoiler@recode#1#2{%
	\def\spoiler@tmp{#2}%
	\catcode`#1=\active%
	\lccode`\~=`#1%
	\lowercase{\let~\spoiler@tmp}}
\def\spoiler@end#1#2#3{\end{Spoiler}}%

	\let\e=\spoiler@end%
